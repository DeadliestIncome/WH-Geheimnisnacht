	id = DR_startup.9
	is_triggered_only = yes
	hide_window = yes

	trigger = {
		has_racial_trait = no
	}
	
	immediate = {
		if = {
			limit = {
				has_racial_trait = no
			}
			assign_racial_trait = yes
		}
		location = {
			owner = {
				if = {
					limit = {
						has_racial_trait = no
					}
					assign_racial_trait = yes
				}
				any_courtier = {
					limit = {
						has_racial_trait = no
					}
					assign_racial_trait = yes
				}
				any_vassal = {
					limit = {

					}

						limit = {
							has_racial_trait = no
						}
						assign_racial_trait = yes
					}
				}
			}
		}
	}

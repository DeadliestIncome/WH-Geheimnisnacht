z_plot_fail = {
	OR = {
		has_character_modifier = z_foresight_blessing_plot
		has_character_modifier = z_divination_plot_block
	}

}

z_plot_succeed = {
	OR = {
		AND = {
			FROMFROM = {
				has_character_modifier = z_divination_plot_success
			}
			NOT = {
				OR = {
					has_character_modifier = z_foresight_blessing_plot
					has_character_modifier = z_divination_plot_block
				}
			}
		}
		has_character_modifier = z_foresight_curse_plot
	}
}
